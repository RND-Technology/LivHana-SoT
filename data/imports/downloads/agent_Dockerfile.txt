FROM python:3.12-slim

ENV WORKDIR /app
WORKDIR $WORKDIR

# Install build tools, audio tools (PortAudio, ALSA, FFmpeg, eSpeak), and git/curl
RUN apt-get update && apt-get install -y --no-install-recommends \
    build-essential \
    git \
    curl \
    portaudio19-dev \
    alsa-utils \
    espeak-ng \
    ffmpeg \
    && rm -rf /var/lib/apt/lists/*

COPY requirements.txt .
RUN pip install --no-cache-dir -r requirements.txt

COPY . .

CMD ["python", "main.py"]
